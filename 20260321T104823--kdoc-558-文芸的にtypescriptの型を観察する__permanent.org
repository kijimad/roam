* 概要

TypeScriptでは、エディタ上で、型が変動していく(ナローイング)様子を観察できる。これを文芸的に表現したいと考えたとき、 ~twoslash~ 記法というものが使えるようである。

Emacsのorg-modeで使えるようにした。

- [[https://github.com/kijimad/ob-twoslash][kijimad/ob-twoslash]]

#+begin_src twoslash
  const a: string | null = "a"
  //    ^?
  if (a !== null) {
     a
  // ^?
  }
#+end_src

#+RESULTS:
#+begin_src
const a: string | null (line 1)
const a: string (line 3)
#+end_src

#+begin_src twoslash
  type Vec3D = Record<'X' | 'Y' | 'Z', number>;
  //   ^?
#+end_src

#+RESULTS:
#+begin_src
type Vec3D = {
    X: number;
    Y: number;
    Z: number;
} (line 1)
#+end_src
