概要

各桁の合計をスマートに得る方法。

% 10 で右端の桁の数字を得ること、 /= で右端の桁の数字を落とすのがポイントに見える。 

package main

import "fmt"

func sum(target int) int {
        result := 0
        for target > 0 {
                result += target % 10
                target /= 10
        }
        return result
}

func main() {
        fmt.Println(sum(1024))
        fmt.Println(sum(55))
        fmt.Println(sum(5555))
}
