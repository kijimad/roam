* 概要

TypeScriptがJavaScriptに変換される様子を見る。

#+caption: TypeScriptをJavaScriptに変換してみる。型の記載はなくなっているのを確認できる
#+begin_src shell
  cat << 'EOF' | npx -y esbuild --loader=ts --format=cjs 2>&1
  /* 変数定義 */
  let color: string = "red";

  /* 関数定義と呼び出し */
  function greet1(name: string) {}
  greet1("hello")

  /* 三項演算子 */
  1 ? "a" : "b"

  /* 型定義 */
  type animal = "dog" | "cat"

  EOF
#+end_src

#+RESULTS:
#+begin_src
let color = "red";
function greet1(name) {
}
greet1("hello");
1 ? "a" : "b";
#+end_src
