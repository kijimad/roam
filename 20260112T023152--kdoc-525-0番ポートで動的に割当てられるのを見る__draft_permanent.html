概要

0番ポートを指定すると動的にポートが割り当てられるのを確かめた。 

import (
  "net"
  "fmt"
  "github.com/stretchr/testify/require"
)

func TestPort(t *testing.T) {
  listenHelper(t)
  listenHelper(t)
  listenHelper(t)
}

func listenHelper(t *testing.T) {
   t.Helper()

   l, err := net.Listen("tcp", ":0")
   require.NoError(t, err)

   fmt.Println(l.Addr())
}
